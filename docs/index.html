function setup() sb = {} -- Make a text screen t = Text.new(50, 30) -- Make a 50x30 text screen table.insert(sb, t) -- Add its reference to the "sb" table. t:color(0, 15) -- Make the text transparent on a background of white. t:print("This is an example script, showing off the text layer, sprites, and a background!") -- Make a SmileBASIC logo s = Sprites.new(4095) -- Make a sprite table.insert(sb, s) -- Add its reference to the "sb" table. -- Make 64 sprites for i = 0, 63 do s = Sprites.new(i) -- Make a sprite table.insert(sb, s) -- Add its reference to the "sb" table. s:offset((i % 16) * 16, math.floor(i / 16) * 16) -- Show it in a grid s:home(8, 8) -- Set sprite's home to center of sprite end end -- Every frame, do this function update() -- We didn't lose the text reference, so print some text every frame t:print("Hello, world! ") -- Handle callbacks and animations, neither of which you're currently using. Still useful. for i = 1, #sb do sb[i]:update() end end function draw() love.graphics.clear(0, 0, 0) -- Z-sort everything you added ZSort.clear() for i = 1, #sb do ZSort.add(sb[i]) end ZSort.flush() end
